# install vite project
 - npm create vite@latest
 - cd project_name
 - npm install 

 # install required packages
 - npm install pinia
 - npm install vue-router@4
 - npm install @vueuse/core
 
 # after installing packages : run the project
 - npm run dev
 - goto http://localhost:5173/
